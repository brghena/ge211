11 static T weighted_average(T t, double weight, U u) noexcept
12 {
13  auto f1 = static_cast<double>(t);
14  auto f2 = static_cast<double>(u);
15  double result = (1 - weight) * f1 + weight * f2;

37 // Creates a color from the HSL/HSV-style hue, the chroma, an adjustment
38 // for brightness, and the alpha.
39 //
40 // From https://en.wikipedia.org/wiki/HSL_and_HSV
41 static Color from_hcma(double hue,
42  double C,
43  double m,
44  double alpha) noexcept
45 {
46  double H6 = std::fmod(hue, 360.0) / 60.0;
47  double X = C * (1 - std::fabs(std::fmod(H6, 2) - 1));
48 
49  double r1 = 0, g1 = 0, b1 = 0;
50 
51  if (H6 <= 1) {
52  r1 = C;
53  g1 = X;
54  } else if (H6 <= 2) {
55  r1 = X;
56  g1 = C;
57  } else if (H6 <= 3) {
58  g1 = C;
59  b1 = X;
60  } else if (H6 <= 4) {
61  g1 = X;
62  b1 = C;
63  } else if (H6 <= 5) {
64  b1 = C;
65  r1 = X;
66  } else {
67  b1 = X;
68  r1 = C;
69  }
70 
71  return Color::from_rgba(r1 + m, g1 + m, b1 + m, alpha);
72 }

75  double alpha) noexcept
76 {
77  double C = (1 - std::fabs(2 * lightness - 1)) * saturation;

130 static std::tuple<double, double, double, double> to_HCMm(Color color) noexcept
131 {
132  double R = color.red() / 255.0;
133  double G = color.green() / 255.0;
134  double B = color.blue() / 255.0;
135 
136  double M = std::max(R, std::max(G, B));
137  double m = std::min(R, std::min(G, B));
138  double C = M - m;
139 
140  double H6 =
141  (M == R) ? std::fmod((G - B) / C, 6) :
142  (M == G) ? (B - R) / C + 2 :
143  (R - G) / C + 4;
144 
145  double H = 60 * H6;
146 
147  return std::make_tuple(H, C, M, m);
148 }

150 Color::HSLA Color::to_hsla() const noexcept
151 {
152  double H, C, M, m;
153  std::tie(H, C, M, m) = to_HCMm(*this);
154 
155  double L = (M + m) / 2;
156  double S = (L == 1) ? 0 : C / (1 - std::fabs(2 * L - 1));
157 
158  return {H, S, L, alpha() / 255.0};
159 }

162 {

164  std::tie(H, C, M, m) = to_HCMm(*this);
165
